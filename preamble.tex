% Add custom fonts
\usepackage{fontspec}
    \setmainfont{minionpro}[
        Path = fonts/,
        Extension = .otf,
        UprightFont = *-regular,
        ItalicFont = *-it,
        BoldFont = *-semibold,
        BoldItalicFont = *-semiboldit
    ]

% need this package to add figures
\usepackage{graphicx}

% need to add this because the default captions are very close to the table
\usepackage{caption}

% Need to define the build/ dir as the output dir of the package
\usepackage{minted}
\usemintedstyle{tango}

% Change listings names for code snippets
\renewcommand\listingscaption{Code Snippet}
\renewcommand\listoflistingscaption{List of Code Snippets}

% Package to create nice headers and footers
\usepackage{fancyhdr}
\pagestyle{fancy}

% add horizontal line on header
\fancyhf{}

% name of the chapter on the left of the header
\lhead{\leftmark}

% name of the section on the right of the header
\rhead{\rightmark}

% add the page number at the right of the footer
\rfoot{\thepage}
 
% Add hyperlinks in toc to sections and inside text where you cite a reference
% or you refer to a table, listing or figure
\usepackage[hidelinks]{hyperref}

% for the bibliography
\usepackage[language=english, backend=biber]{biblatex}
\addbibresource{references.bib}
